    <script>
        function switchTab(e, tabId) {
            // Hide all tabs
            document.querySelectorAll('.tab-content').forEach(el => el.classList.add('hidden'));
            
            // Reset all buttons
            const buttons = e.target.parentElement.querySelectorAll('button');
            buttons.forEach(btn => {
                btn.className = "px-4 py-2 text-sm font-medium text-muted hover:text-white transition-colors";
            });
            
            // Activate current button
            e.target.className = "px-4 py-2 text-sm font-medium text-white border-b-2 border-primary transition-colors";
            
            // Show content
            document.getElementById(tabId).classList.remove('hidden');
        }

        function copyLink(elementId) {
            const element = document.getElementById(elementId);
            copyToClipboard(element.value);
        }

        function copyToClipboard(text) {
            if (!text) return;
            navigator.clipboard.writeText(text).then(() => {
                showToast();
            }).catch(() => {
                alert('Failed to copy');
            });
        }

        function showToast() {
            const toast = document.getElementById('toast');
            toast.classList.remove('translate-y-20', 'opacity-0');
            setTimeout(() => {
                toast.classList.add('translate-y-20', 'opacity-0');
            }, 2000);
        }

        function toggleConfig() {
            const content = document.getElementById('configContent');
            const arrow = document.getElementById('arrow');
            content.classList.toggle('hidden');
            if (content.classList.contains('hidden')) {
                arrow.style.transform = 'rotate(0deg)';
            } else {
                arrow.style.transform = 'rotate(180deg)';
            }
        }
    </script>
